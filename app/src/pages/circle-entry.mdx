## 1. お支払い手続き

以下の購入フォームよりお支払い手続きを行ってください。
決済代行会社としてPayPalを利用しています。お支払い手続きは必ず1度のみ行ってください。

- お支払いは必ず1回のみ行ってください
- 出納業務簡略化のため、今回より支払い方法をPayPalのみに変更しています
- 運営事務局よりお支払いを免除されたサークルはこの手順をスキップしてください

<script>
  function initPayPalButton() {
    paypal.Buttons({
      style: {
        shape: 'rect',
        color: 'blue',
        layout: 'vertical',
        label: 'pay',
        
      },

      createOrder: function(data, actions) {
        return actions.order.create({
          purchase_units: [{"description":"技書博7 サークル出展料","amount":{"currency_code":"JPY","value":8000}}]
        });
      },

      onApprove: function(data, actions) {
        return actions.order.capture().then(function(orderData) {
          // Full available details
          console.log('Capture result', orderData, JSON.stringify(orderData, null, 2));
          // Show a success message within this page, e.g.
          const element = document.getElementById('paypal-button-container');
          element.innerHTML = '';
          element.innerHTML = '<h3>決済が完了しました。続けて出展登録してください。</h3>';
        });
      },

      onError: function(err) {
        console.log(err);
      }
    }).render('#paypal-button-container')
  }
  initPayPalButton();
</script>
